# Chat Visual Feedback + Esc Abort Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add visual feedback to the JARVIS chat panel (tool execution bars, streaming cursor, abort indicators) and emit the bus events that feed them.

**Architecture:** Extend the `AIStreamMessage` event type with tool lifecycle and abort events. JarvisCore emits these on the bus. ChatPiece broadcasts them to SSE clients. ChatOutput renders the new states with CSS animations.

**Tech Stack:** TypeScript, React, Server-Sent Events, CSS animations.

  abortSession(sessionId: string): void {
    const managed = this.sessions.get(sessionId);
    if (!managed || managed.state === "idle") return;

    const wasWaitingTools = managed.state === "waiting_tools";
    log.info({ sessionId, state: managed.state }, "JarvisCore: abort requested");
    this.sessions.abort(sessionId);
    this.pendingPrompts.delete(sessionId);
    this.state = "online";
    this.updateHud();

    // Emit tool_cancelled if we were waiting for tools
    if (wasWaitingTools && managed.pendingToolCalls) {
      for (const tc of managed.pendingToolCalls) {
        this.bus.publish({
          channel: "ai.stream",
          source: "jarvis-core",
          target: sessionId,
          event: "tool_cancelled",
          toolName: tc.name,
          toolId: tc.id,
        });
      }
    }

    this.bus.publish({
      channel: "ai.stream",
      source: "jarvis-core",
      target: sessionId,
      event: "aborted",
    });
  }
```

    this.bus.subscribe<AIStreamMessage>("ai.stream", (msg) => {
      if (msg.target?.startsWith("actor-")) return; // actors have own chat
      const source = msg.source === "jarvis-core" ? "jarvis" : msg.source;
      switch (msg.event) {
        case "delta":
          this.broadcast({ type: "delta", text: msg.text, source, session: msg.target });
          break;
        case "complete":
          this.broadcast({ type: "done", fullText: msg.text ?? "", source, session: msg.target });
          break;
        case "error":
          this.broadcast({ type: "error", error: msg.error, source, session: msg.target });
          break;
        case "tool_start":
          this.broadcast({ type: "tool_start", name: msg.toolName, id: msg.toolId, source, session: msg.target });
          break;
        case "tool_done":
          this.broadcast({ type: "tool_done", name: msg.toolName, id: msg.toolId, ms: msg.toolMs, source, session: msg.target });
          break;
        case "tool_cancelled":
          this.broadcast({ type: "tool_cancelled", name: msg.toolName, id: msg.toolId, source, session: msg.target });
          break;
        case "aborted":
          this.broadcast({ type: "aborted", source, session: msg.target });
          break;
      }
    });
```

### Task 4: Update ChatOutput to render tool bars, streaming cursor, and abort indicator

**Files:**
- Modify: `app/ui/src/components/panels/ChatOutput.tsx`

- [ ] **Step 1: Rewrite ChatOutput with new state handling**

Replace the entire `ChatOutput.tsx`:

```tsx
import { useState, useRef, useEffect } from 'react'

type Message = {
  role: 'user' | 'jarvis'
  text: string
  source?: string
  session?: string
  aborted?: boolean
}

type ActiveTool = {
  name: string
  id: string
  status: 'running' | 'done' | 'cancelled'
  startedAt: number
  ms?: number
}

export function ChatOutput() {
  const [messages, setMessages] = useState<Message[]>([])
  const [streamingText, setStreamingText] = useState('')
  const [tools, setTools] = useState<ActiveTool[]>([])
  const [isStreaming, setIsStreaming] = useState(false)
  const containerRef = useRef<HTMLDivElement>(null)

  useEffect(() => {
    const el = containerRef.current
    if (el) el.scrollTop = el.scrollHeight
  }, [messages, streamingText, tools])

  useEffect(() => {
    const source = new EventSource('/chat-stream')

    source.onmessage = (event) => {
      const data = JSON.parse(event.data)

      switch (data.type) {
        case 'user':
          setMessages(prev => [...prev, { role: 'user', text: data.text, source: data.source, session: data.session }])
          setTools([])
          break
        case 'delta':
          setIsStreaming(true)
          setStreamingText(prev => prev + data.text)
          break
        case 'done':
          setIsStreaming(false)
          if (data.fullText) {
            setMessages(prev => [...prev, { role: 'jarvis', text: data.fullText, source: data.source, session: data.session }])
          }
          setStreamingText('')
          setTools([])
          break
        case 'error':
          setIsStreaming(false)
          setMessages(prev => [...prev, { role: 'jarvis', text: `[Error: ${data.error}]`, source: data.source }])
          setStreamingText('')
          setTools([])
          break
        case 'tool_start':
          setTools(prev => [...prev, { name: data.name, id: data.id, status: 'running', startedAt: Date.now() }])
          break
        case 'tool_done':
          setTools(prev => prev.map(t =>
            t.id === data.id ? { ...t, status: 'done' as const, ms: Date.now() - t.startedAt } : t
          ))
          break
        case 'tool_cancelled':
          setTools(prev => prev.map(t =>
            t.id === data.id ? { ...t, status: 'cancelled' as const } : t
          ))
          break
        case 'aborted':
          setIsStreaming(false)
          if (streamingText) {
            setMessages(prev => [...prev, { role: 'jarvis', text: streamingText, source: data.source, session: data.session, aborted: true }])
            setStreamingText('')
          }
          break
      }
    }

    return () => source.close()
  }, [])

  const labelFor = (msg: Message) => {
    if (msg.role === 'user') {
      if (!msg.source || msg.source === 'chat') return 'YOU'
      return msg.source.toUpperCase()
    }
    if (msg.source && msg.source !== 'jarvis') return msg.source.toUpperCase()
    return 'JARVIS'
  }

  const labelColor = (msg: Message) => {
    if (msg.role === 'user') {
      if (!msg.source || msg.source === 'chat') return 'var(--chat-user-label)'
      if (msg.source === 'system') return '#fa4'
      return '#4af'
    }
    if (msg.source && msg.source !== 'jarvis') return '#a6f'
    return 'var(--chat-jarvis-label)'
  }

  const msgStyle = (msg: Message): React.CSSProperties => ({
    color: msg.role === 'user' ? 'var(--chat-user)' : 'var(--chat-jarvis)',
    whiteSpace: 'pre-wrap',
    wordBreak: 'break-word',
    lineHeight: '1.5',
    marginBottom: '4px',
  })

  return (
    <div ref={containerRef} style={{ height: '100%', overflowY: 'auto', padding: '8px 12px', fontSize: '11px', fontFamily: 'var(--font-mono)' }}>
      {messages.map((msg, i) => (
        <div key={i} style={msgStyle(msg)}>
          <span style={{ color: labelColor(msg), marginRight: '8px', fontFamily: 'var(--font-display)', fontSize: '9px', fontWeight: 600 }}>
            {labelFor(msg)}
          </span>
          {msg.text}
          {msg.aborted && (
            <span style={{ color: '#666', fontStyle: 'italic', marginLeft: '8px', fontSize: '10px' }}>⊘ interrupted</span>
          )}
        </div>
      ))}

      {tools.length > 0 && (
        <div style={{ marginBottom: '4px' }}>
          {tools.map(t => (
            <div key={t.id} style={{
              padding: '3px 8px',
              marginBottom: '2px',
              borderRadius: '4px',
              fontSize: '10px',
              borderLeft: t.status === 'running' ? '3px solid #f1fa8c'
                : t.status === 'done' ? '3px solid #50fa7b'
                : '3px solid #666',
              background: '#1a1e2e',
              color: t.status === 'running' ? '#f1fa8c'
                : t.status === 'done' ? '#50fa7b'
                : '#666',
            }}>
              {t.status === 'running' && <span style={{ animation: 'pulse 1.5s infinite' }}>⚡</span>}
              {t.status === 'done' && '✓'}
              {t.status === 'cancelled' && '⊘'}
              {' '}<strong>{t.name}</strong>
              {t.status === 'done' && t.ms != null && ` ${t.ms}ms`}
              {t.status === 'cancelled' && ' cancelled'}
            </div>
          ))}
        </div>
      )}

      {streamingText && (
        <div style={{ color: 'var(--chat-jarvis)', whiteSpace: 'pre-wrap', lineHeight: '1.5', marginBottom: '4px' }}>
          <span style={{ color: 'var(--chat-jarvis-label)', marginRight: '8px', fontFamily: 'var(--font-display)', fontSize: '9px', fontWeight: 600 }}>JARVIS</span>
          {streamingText}
          {isStreaming && <span className="streaming-cursor" />}
        </div>
      )}

      <style>{`
        @keyframes pulse { 0%,100%{opacity:1} 50%{opacity:0.4} }
        @keyframes blink { 0%,100%{opacity:1} 50%{opacity:0} }
        .streaming-cursor {
          display: inline-block;
          width: 7px;
          height: 12px;
          background: var(--chat-jarvis-label, #50fa7b);
          animation: blink 1s infinite;
          vertical-align: middle;
          margin-left: 2px;
        }
      `}</style>
    </div>
  )
}
```

**Goal:** Replace the static HUD with a data-driven renderer where the backend controls all visual state via a `/hud` endpoint.

**Architecture:** Backend exposes `HudState` (reactor status + mind-map nodes + panel visibility). Frontend is a pure renderer — fetches state every 2s and renders. No frontend business logic.

```typescript
// ui/src/components/ReactorCore.tsx
import type { HudReactor } from '../types/hud'

const statusColors: Record<string, string> = {
  online: '#4af',
  processing: '#fa4',
  loading: '#a6f',
  initializing: '#a6f',
  offline: '#f44',
}

export function ReactorCore({ reactor, size = 200 }: { reactor: HudReactor; size?: number }) {
  const color = statusColors[reactor.status] ?? '#f44'
  const isActive = reactor.status !== 'offline'
  const isWorking = reactor.status === 'processing' || reactor.status === 'loading'
  const cx = size / 2
  const cy = size / 2

  return (
    <svg width={size} height={size} viewBox={`0 0 ${size} ${size}`}>
      {/* Outer ring */}
      <circle cx={cx} cy={cy} r={cx * 0.9} fill="none" stroke={color} strokeWidth="1.5"
        opacity="0.25" strokeDasharray="6 4"
        style={isWorking ? { animation: 'spin 8s linear infinite', transformOrigin: 'center' } : undefined}
      />

      {/* Middle ring */}
      <circle cx={cx} cy={cy} r={cx * 0.7} fill="none" stroke={color} strokeWidth="1"
        opacity="0.4" strokeDasharray="3 3"
        style={isWorking ? { animation: 'spin-reverse 6s linear infinite', transformOrigin: 'center' } : undefined}
      />

      {/* Inner ring */}
      <circle cx={cx} cy={cy} r={cx * 0.5} fill="none" stroke={color} strokeWidth="1.5" opacity="0.6" />

      {/* Core glow */}
      <circle cx={cx} cy={cy} r={cx * 0.35} fill={`url(#core-glow-${reactor.status})`}
        opacity={isActive ? 0.9 : 0.3}
        style={isActive ? {
          animation: isWorking ? 'pulse-fast 0.8s ease-in-out infinite' : 'pulse-slow 2s ease-in-out infinite',
        } : undefined}
      />

      {/* Core bright center */}
      <circle cx={cx} cy={cy} r={cx * 0.15} fill={`url(#core-bright-${reactor.status})`} />

      {/* Spokes */}
      {[0, 90, 180, 270].map(deg => {
        const rad = (deg * Math.PI) / 180
        const x1 = cx + Math.cos(rad) * cx * 0.52
        const y1 = cy + Math.sin(rad) * cx * 0.52
        const x2 = cx + Math.cos(rad) * cx * 0.88
        const y2 = cy + Math.sin(rad) * cx * 0.88
        return <line key={deg} x1={x1} y1={y1} x2={x2} y2={y2} stroke={color} strokeWidth="0.5" opacity="0.2" />
      })}

      {/* Core label */}
      <text x={cx} y={cy - 4} textAnchor="middle" fill="#fff" fontSize={size * 0.055}
        fontFamily="Orbitron, sans-serif" fontWeight="500">
        {reactor.coreLabel}
      </text>
      <text x={cx} y={cy + size * 0.06} textAnchor="middle" fill={color} fontSize={size * 0.04}
        fontFamily="'JetBrains Mono', monospace" opacity="0.7">
        {reactor.coreSubLabel}
      </text>

      {/* Gradients */}
      <defs>
        <radialGradient id={`core-glow-${reactor.status}`}>
          <stop offset="0%" stopColor={color} stopOpacity="0.5" />
          <stop offset="100%" stopColor={color} stopOpacity="0" />
        </radialGradient>
        <radialGradient id={`core-bright-${reactor.status}`}>
          <stop offset="0%" stopColor="#fff" stopOpacity="0.35" />
          <stop offset="50%" stopColor={color} stopOpacity="0.2" />
          <stop offset="100%" stopColor={color} stopOpacity="0" />
        </radialGradient>
      </defs>
    </svg>
  )
}
```

```typescript
// ui/src/components/MindMapNodes.tsx
import type { HudNode } from '../types/hud'

type Props = {
  nodes: HudNode[];
  cx: number;
  cy: number;
  innerRadius: number;
  outerRadius: number;
}

export function MindMapNodes({ nodes, cx, cy, innerRadius, outerRadius }: Props) {
  if (nodes.length === 0) return null

  return (
    <g>
      {nodes.map((node, i) => {
        const angle = (i / nodes.length) * Math.PI * 2 - Math.PI / 2
        const edgeX = cx + Math.cos(angle) * innerRadius
        const edgeY = cy + Math.sin(angle) * innerRadius
        const labelX = cx + Math.cos(angle) * outerRadius
        const labelY = cy + Math.sin(angle) * outerRadius

        return (
          <g key={node.id}>
            {/* Connector line */}
            <line x1={edgeX} y1={edgeY} x2={labelX} y2={labelY}
              stroke={node.color} strokeWidth="0.5" opacity="0.3" />

            {/* Endpoint dot */}
            <circle cx={labelX} cy={labelY} r="2" fill={node.color} opacity="0.5" />

            {/* Label */}
            <text x={labelX} y={labelY - 10} textAnchor="middle"
              fill="#445" fontSize="8" fontFamily="Orbitron, sans-serif"
              letterSpacing="1">
              {node.label}
            </text>

            {/* Value */}
            <text x={labelX} y={labelY + 4} textAnchor="middle"
              fill={node.color} fontSize="13" fontFamily="'JetBrains Mono', monospace"
              fontWeight="500"
              style={node.pulse ? { animation: 'glow-pulse 1s ease-in-out infinite', color: node.color } : undefined}>
              {node.value}
            </text>

            {/* Children (if expanded) */}
            {node.expanded && node.children?.map((child, ci) => {
              const childAngle = angle + ((ci - (node.children!.length - 1) / 2) * 0.3)
              const childRadius = outerRadius * 1.4
              const childX = cx + Math.cos(childAngle) * childRadius
              const childY = cy + Math.sin(childAngle) * childRadius

              return (
                <g key={child.id}>
                  <line x1={labelX} y1={labelY} x2={childX} y2={childY}
                    stroke={child.color} strokeWidth="0.3" opacity="0.2" />
                  <text x={childX} y={childY - 6} textAnchor="middle"
                    fill="#445" fontSize="7" fontFamily="Orbitron, sans-serif">
                    {child.label}
                  </text>
                  <text x={childX} y={childY + 4} textAnchor="middle"
                    fill={child.color} fontSize="10" fontFamily="'JetBrains Mono', monospace">
                    {child.value}
                  </text>
                </g>
              )
            })}
          </g>
        )
      })}
    </g>
  )
}
```

```typescript
// ui/src/components/panels/ChatPanel.tsx
import { useState, useRef, useEffect, type KeyboardEvent } from 'react'

type Message = { role: 'user' | 'jarvis'; text: string }

export function ChatPanel({ disabled }: { disabled?: boolean }) {
  const [messages, setMessages] = useState<Message[]>([])
  const [input, setInput] = useState('')
  const [loading, setLoading] = useState(false)
  const bottomRef = useRef<HTMLDivElement>(null)
  const inputRef = useRef<HTMLInputElement>(null)

  useEffect(() => {
    bottomRef.current?.scrollIntoView({ behavior: 'smooth' })
  }, [messages])

  const send = async () => {
    const prompt = input.trim()
    if (!prompt || loading) return

    setMessages(prev => [...prev, { role: 'user', text: prompt }])
    setInput('')
    setLoading(true)

    try {
      const res = await fetch('/chat', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ prompt }),
      })
      const data = await res.json()
      setMessages(prev => [...prev, { role: 'jarvis', text: data.result }])
    } catch {
      setMessages(prev => [...prev, { role: 'jarvis', text: '[Error: could not reach Jarvis]' }])
    } finally {
      setLoading(false)
      inputRef.current?.focus()
    }
  }

  const handleKey = (e: KeyboardEvent) => {
    if (e.key === 'Enter' && !e.shiftKey) {
      e.preventDefault()
      send()
    }
  }

  return (
    <div style={{
      flex: 1,
      display: 'flex',
      flexDirection: 'column',
      border: '1px solid #1a2a3a',
      borderRadius: '4px',
      overflow: 'hidden',
      WebkitAppRegion: 'no-drag' as any,
    }}>
      <div style={{
        flex: 1,
        overflowY: 'auto',
        padding: '8px 12px',
        display: 'flex',
        flexDirection: 'column',
        gap: '6px',
        fontSize: '11px',
      }}>
        {messages.map((msg, i) => (
          <div key={i} style={{ color: msg.role === 'user' ? '#6a8' : '#4af' }}>
            <span style={{ color: '#445', marginRight: '6px', fontFamily: 'Orbitron, sans-serif', fontSize: '8px' }}>
              {msg.role === 'user' ? 'YOU' : 'JARVIS'}
            </span>
            {msg.text}
          </div>
        ))}
        {loading && (
          <div style={{ color: '#fa4' }}>
            <span style={{ color: '#445', marginRight: '6px', fontFamily: 'Orbitron, sans-serif', fontSize: '8px' }}>JARVIS</span>
            ...
          </div>
        )}
        <div ref={bottomRef} />
      </div>

      <div style={{
        borderTop: '1px solid #1a2a3a',
        padding: '6px 12px',
        display: 'flex',
        gap: '8px',
        alignItems: 'center',
      }}>
        <span style={{ color: '#445', fontSize: '8px', fontFamily: 'Orbitron, sans-serif' }}>YOU</span>
        <input
          ref={inputRef}
          value={input}
          onChange={e => setInput(e.target.value)}
          onKeyDown={handleKey}
          disabled={disabled}
          placeholder={disabled ? 'Offline' : 'Type a message...'}
          autoFocus
          style={{
            flex: 1,
            background: 'transparent',
            border: 'none',
            outline: 'none',
            color: '#6a8',
            fontSize: '11px',
          }}
        />
      </div>
    </div>
  )
}
```

```typescript
// ui/src/App.tsx
import { useState, useEffect } from 'react'
import type { HudState } from './types/hud'
import { HudRenderer } from './components/HudRenderer'

const DEFAULT_STATE: HudState = {
  layout: 'center-chat-bottom',
  reactor: { status: 'offline', coreLabel: 'CONNECTING', coreSubLabel: '...' },
  nodes: [],
  panels: [],
}

export function App() {
  const [state, setState] = useState<HudState>(DEFAULT_STATE)

  useEffect(() => {
    const poll = async () => {
      try {
        const res = await fetch('/hud')
        if (res.ok) {
          setState(await res.json())
        }
      } catch {
        setState(DEFAULT_STATE)
      }
    }

    poll()
    const interval = setInterval(poll, 2000)
    return () => clearInterval(interval)
  }, [])

  return <HudRenderer state={state} />
```
